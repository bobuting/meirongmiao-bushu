```typescript
export interface UnifiedReverseDeps {
  // Video processing
  resolveVideoUrl: (inputUrl: string) => Promise<string>;
  downloadVideoForLlm: (url: string) => Promise<{ base64: string; mimeType: string } | null>;

  // LLM
  resolveProvider: (routeKeys: ProviderRouteKey[]) => Promise<VideoHotTrendResolvedProvider | null>;
  callLlm: (
    provider: VideoHotTrendResolvedProvider,
    prompt: string,
    video: { base64: string; mimeType: string },
    timeoutMs: number
  ) => Promise<VideoHotTrendLlmPlainTextResult>;

  // Audit
  createAuditRecord: (input: {
    routeKey: ProviderRouteKey;
    businessContext: string;
    projectId?: string;
    userId?: string;
    messages: Array<{ role: string; content: string }>;
    modelParams: { temperature?: number; max_tokens?: number; top_p?: number };
    provider: VideoHotTrendResolvedProvider;
  }) => { auditId: string; startedAt: number };
  finalizeAuditSuccess: (input: {
    auditId: string;
    startedAt: number;
    actualModel: string;
    responseText: string;
    inputTokens?: number;
    outputTokens?: number;
    ttftMs?: number;
  }) => void;
  finalizeAuditError: (input: {
    auditId: string;
    startedAt: number;
    errorCode: string;
    errorMessage: string;
  }) => void;

  // Utilities
  extractJsonValue: (text: string) => unknown | null;
  log: { info: (obj: unknown, msg?: string) => void; warn: (obj: unknown, msg?: string) => void };
  generateId: () => string;
  now: () => number;
}
```
