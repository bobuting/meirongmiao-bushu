```typescript
export async function runCoreReversePipeline(
  deps: UnifiedReverseDeps,
  input: CoreReverseInput,
): Promise<CoreReverseOutput> {
  // 阶段A: 解析视频 URL（如果需要）
  const resolvedVideoUrl = await deps.resolveVideoUrl(input.videoUrl);

  // 阶段B: 下载视频到内存（base64）
  const downloadResult = await deps.downloadVideoForLlm(resolvedVideoUrl);
  if (!downloadResult) {
    return {
      rawLlmOutput: null,
      resolvedVideoUrl,
      success: false,
      errorCode: CORE_REVERSE_ERROR_CODES.VIDEO_DOWNLOAD_FAILED,
      errorMessage: "视频下载失败",
    };
  }

  // 阶段C: 解析 LLM provider（使用调用方传入的 routeKeys）
  const provider = await deps.resolveProvider(input.routeKeys);
  if (!provider) {
    return {
      rawLlmOutput: null,
      resolvedVideoUrl,
      success: false,
      errorCode: CORE_REVERSE_ERROR_CODES.NO_PROVIDER,
      errorMessage: "无可用的 LLM 提供者",
    };
  }

  // 阶段D: 构建 prompt
  const prompt = await buildVideoStoryboardPrompt({
    videoUrl: resolvedVideoUrl,
    topicLabel: input.topicLabel ?? "视频",
    topicId: input.topicId ?? deps.generateId(),
  });
  const promptContent = await getPromptContent(PROMPT_CODE_VIDEO_STORYBOARD_ANALYSIS, {
    userPrompt: prompt,
  });

  // 阶段E: 创建审计记录
  const auditRecord = deps.createAuditRecord({
    routeKey: input.auditContext.routeKey,
    businessContext: input.auditContext.businessContext,
    projectId: input.auditContext.projectId,
    userId: input.auditContext.userId,
    messages: [{ role: "user", content: prompt }],
    modelParams: { temperature: 0.3 },
    provider,
  });

  try {
    // 阶段F: 调用 LLM（统一方法，内部处理 Gemini/OpenAI dispatch）
    const timeoutMs = Math.max(provider.timeoutMs, 180_000);
    const llmResult = await deps.callLlm(
      provider,
      promptContent.prompt,
      { base64: downloadResult.base64, mimeType: downloadResult.mimeType },
      timeoutMs,
    );

    // 阶段G: 解析 JSON
    const parsed = deps.extractJsonValue(llmResult.text);
    if (!parsed || typeof parsed !== "object") {
      deps.finalizeAuditError({
        auditId: auditRecord.auditId,
        startedAt: auditRecord.startedAt,
        errorCode: CORE_REVERSE_ERROR_CODES.LLM_RESPONSE_INVALID,
        errorMessage: "LLM 响应无效，无法解析为 JSON",
      });
      return {
        rawLlmOutput: null,
        resolvedVideoUrl,
        success: false,
        errorCode: CORE_REVERSE_ERROR_CODES.LLM_RESPONSE_INVALID,
        errorMessage: "LLM 响应无效，无法解析为 JSON",
      };
    }

    // 阶段H: 标准化输出
    const output = normalizeLlmReverseOutput(parsed);

    // 完成审计记录（成功）
    deps.finalizeAuditSuccess({
      auditId: auditRecord.auditId,
      startedAt: auditRecord.startedAt,
      actualModel: provider.model,
      responseText: llmResult.text,
    });

    return {
      rawLlmOutput: output,
      resolvedVideoUrl,
      success: true,
      errorCode: null,
      errorMessage: null,
    };
  } catch (error) {
    const errorCode = CORE_REVERSE_ERROR_CODES.LLM_CALL_FAILED;
    const errorMessage = error instanceof Error ? error.message : String(error);

    deps.finalizeAuditError({
      auditId: auditRecord.auditId,
      startedAt: auditRecord.startedAt,
      errorCode,
      errorMessage,
    });

    return {
      rawLlmOutput: null,
      resolvedVideoUrl,
      success: false,
      errorCode,
      errorMessage,
    };
  }
}
```

**关键设计点**：
1. 返回结构化结果 CoreReverseOutput，不抛异常（per D-04）
2. resolveProvider 使用 input.routeKeys 参数，由调用方决定 fallback chain（per Pitfall 4）
3. callLlm 统一方法接收 base64/mimeType，适配器内部处理 Gemini/OpenAI dispatch
4. 审计方法使用 createAuditRecord/finalizeAuditSuccess/Error 统一接口
