### Task 1: 创建数据库表

通过 Node.js 脚本直接创建表（不写迁移文件）。使用 `pg` 模块连接到数据库。

```sql
-- 模特图表（Step 3）
-- 存储 AI 自动生成的模特图及其姿势、背景标签
CREATE TABLE IF NOT EXISTS nrm_model_photos (
  id TEXT PRIMARY KEY,
  project_id TEXT NOT NULL,
  image_url TEXT,
  pose_label TEXT,
  bg_label TEXT,
  is_selected BOOLEAN DEFAULT false,
  status TEXT DEFAULT 'pending',
  error_message TEXT,
  sort_order INTEGER DEFAULT 0,
  created_at BIGINT NOT NULL,
  updated_at BIGINT NOT NULL
);
CREATE INDEX IF NOT EXISTS idx_model_photos_project ON nrm_model_photos(project_id);
CREATE INDEX IF NOT EXISTS idx_model_photos_status ON nrm_model_photos(status);

-- 电商详情页区块表（Step 4）
-- 存储 AI 规划的页面结构，每个区块可独立生成图片
CREATE TABLE IF NOT EXISTS nrm_page_sections (
  id TEXT PRIMARY KEY,
  project_id TEXT NOT NULL,
  section_key TEXT NOT NULL,
  section_type TEXT NOT NULL,
  title TEXT,
  goal TEXT,
  copy TEXT,
  visual_prompt TEXT,
  sort_order INTEGER DEFAULT 0,
  status TEXT DEFAULT 'idle',
  current_image_asset_id TEXT,
  editable_data JSONB,
  created_at BIGINT NOT NULL,
  updated_at BIGINT NOT NULL
);
CREATE UNIQUE INDEX IF NOT EXISTS idx_sections_project_key ON nrm_page_sections(project_id, section_key);
CREATE INDEX IF NOT EXISTS idx_sections_project ON nrm_page_sections(project_id);

-- 区块版本历史表（Step 4）
-- 记录每次重新生成时的快照，支持版本切换
CREATE TABLE IF NOT EXISTS nrm_section_versions (
  id TEXT PRIMARY KEY,
  section_id TEXT NOT NULL,
  project_id TEXT NOT NULL,
  version_number INTEGER NOT NULL,
  prompt_snapshot JSONB,
  copy_snapshot JSONB,
  image_asset_id TEXT,
  is_active BOOLEAN DEFAULT false,
  created_at BIGINT NOT NULL
);
CREATE UNIQUE INDEX IF NOT EXISTS idx_versions_section_ver ON nrm_section_versions(section_id, version_number);
CREATE INDEX IF NOT EXISTS idx_versions_section ON nrm_section_versions(section_id);
```

编写 Node.js 脚本创建这些表，放在 `scripts/create-image-project-tables.ts` 或直接通过 `node -e` 执行。
